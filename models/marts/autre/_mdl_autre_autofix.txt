models:
  - name: dim_autre_customers
    description: |
        Customer overview data mart, offering key details for each unique customer. 
        One row per customer. Customer grain mart.
    columns:
      - name: customer_id
        description: The unique key of the orders mart.
        data_tests:
          - not_null
          - unique
        entity:
          type: primary
          name: customer
      - name: customer_name
        description: Customers' full name.
        dimension:
          type: categorical
      - name: count_lifetime_orders
        description: Total number of orders a customer has ever placed.
      - name: first_ordered_at
        description: The timestamp when a customer placed their first order.
        dimension:
          type: time
        granularity: day
      - name: last_ordered_at
        description: The timestamp of a customer's most recent order.
        dimension:
          type: time
        granularity: day
      - name: lifetime_spend_pretax
        description: The sum of all the pre-tax subtotals of every order a customer has placed.
      - name: lifetime_spend
        description: The sum of all the order totals (including tax) that a customer has ever placed.
      - name: customer_type
        description: |
            Options are 'new' or 'returning', indicating if a customer has ordered more than once 
            or has only placed their first order to date.
        data_tests:
          - accepted_values:
              arguments:
                values: ["new", "returning"]
        dimension:
          type: categorical
    semantic_model:
      enabled: true
      name: dim_autre_customers
    agg_time_dimension: first_ordered_at
    
    metrics:
      - name: "customers"
        description: "Count of customers"
        type: simple
        label: "customers"
        agg: count_distinct
        expr: customer_id
      - name: "customers_with_orders"
        label: "customers_with_orders"
        description: "Unique count of customers placing orders"
        type: simple
        agg: count_distinct
        expr: customer_id
      - name: count_lifetime_orders
        description: Total count of orders per customer.
        agg: sum
        type: simple
        hidden: true
      - name: lifetime_spend_pretax
        description: Customer lifetime spend before taxes.
        agg: sum
        type: simple
        hidden: true
      - name: lifetime_spend
        agg: sum
        description: Gross customer lifetime spend inclusive of taxes.
        type: simple
        hidden: true


  - name: order_items
    description: |
      Order items data mart, offering key details for each individual item within an order. 
      One row per order item.
       Items contatined in each order. The grain of the table is one row per order item.

    columns:
      - name: order_item_id
        description: The unique key of the order items mart.
        data_tests:
          - not_null
          - unique
        entity:
          type: primary
          name: order_item
      - name: order_id
        description: The foreign key relating to the order this item belongs to.
        data_tests:
          - relationships:
              arguments:
                to: ref('fct_autre_orders')
                field: order_id
        entity:
          type: foreign
      - name: product_id
        description: The foreign key relating to the product that was ordered.
        entity:
          type: foreign
          name: product
      - name: product_price
        description: The number of this specific product that was ordered.
      - name: supply_cost
        description: The price of a single unit of this product without tax.
      - name: is_food_item
        description: A boolean indicating if this item is a food item.
        dimension:
          type: categorical
      - name: is_drink_item
        description: A boolean indicating if this item is a drink item.
        dimension:
          type: categorical
      - name: ordered_at_day
        description: The timestamp the order containing this item was placed at.


    semantic_model:
      enabled: true
      name: order_items
    agg_time_dimension: ordered_at
    derived_semantics:
      entities: []
      dimensions:
        - name: ordered_at
          type: time
          granularity: day
          expr: date_trunc('day', ordered_at)
    metrics:
      - name: revenue
        description: The revenue generated for each order item. Revenue is calculated as a sum of revenue associated with each product in an order.
        agg: sum
        expr: product_price
        type: simple
        hidden: true
      - name: food_revenue
        description: The revenue generated for each order item. Revenue is calculated as a sum of revenue associated with each product in an order.
        agg: sum
        expr: case when is_food_item = 1 then product_price else 0 end
        type: simple
        hidden: true
      - name: drink_revenue
        description: The revenue generated for each order item. Revenue is calculated as a sum of revenue associated with each product in an order.
        agg: sum
        expr: case when is_drink_item = 1 then product_price else 0 end
        type: simple
        hidden: true


  - name: fct_autre_orders
    description: "Order overview data mart, offering key details for each order inlcluding if it's a customer's first order and a food vs. drink item breakdown. One row per order. Order fact table. This table is at the order grain with one row per order.\n"
    columns:
      - name: order_id
        description: The unique key of the orders mart.
        data_tests:
          - not_null
          - unique
        entity:
          type: primary
      - name: customer_id
        description: The foreign key relating to the customer who placed the order.
        data_tests:
          - relationships:
              arguments:
                to: ref('stg_autre__customers')
                field: customer_id
        entity:
          type: foreign
          name: customer
      - name: location_id
        description: The foreign key relating to the location the order was placed at.
        entity:
          type: foreign
          name: location
      - name: order_total
        description: The total amount of the order in USD including tax.
        dimension:
          type: categorical
          name: order_total_dim
      - name: ordered_at
        description: The timestamp the order was placed at.
      - name: count_food_items
        description: The number of individual food items ordered.
      - name: count_drink_items
        description: The number of individual drink items ordered.
      - name: count_items
        description: The total number of both food and drink items ordered.
      - name: subtotal_food_items
        description: The sum of all the food item prices without tax.
      - name: subtotal_drink_items
        description: The sum of all the drink item prices without tax.
      - name: subtotal
        description: The sum total of both food and drink item prices without tax.
      - name: order_cost
        description: The sum of supply expenses to fulfill the order.
      - name: location_name
        description: The full location name of where this order was placed. Denormalized from `stg_locations`.
      - name: is_first_order
        description: A boolean indicating if this order is from a new customer placing their first order.
      - name: is_food_order
        description: A boolean indicating if this order included any food items.
        dimension:
          type: categorical
      - name: is_drink_order
        description: A boolean indicating if this order included any drink items.

    semantic_model:
      enabled: true
      name: fct_autre_orders
    agg_time_dimension: ordered_at
    derived_semantics:
      entities: []
      dimensions:
        - name: ordered_at
          type: time
          granularity: day
          expr: date_trunc('day', ordered_at)
    metrics:

        description: "Sum of orders value"
        type: simple
        label: "order_total"
        agg: sum
      - name: "locations_with_orders"
        description: "The number of locations that have orders."
        type: simple
        label: "locations_with_orders"
        agg: count_distinct
        expr: location_id
      - name: "order_value_p99"
        description: "The 99th percentile of each order"
        type: simple
        label: "order_value_p99"
        agg: percentile
        expr: order_total
      - name: "discrete_order_value_p99"
        description: "The 99th percentile of each order using a discrete distribution"
        type: simple
        label: "discrete_order_value_p99"
        agg: percentile
        expr: order_total
      - name: large_order
        description: "Count of orders with order total over 20."
        type: simple
        label: "Large Orders"
        filter: |
          {{ Dimension('order_id__order_total_dim') }} >= 20
        agg: sum
        expr: 1
      - name: "food_order_amount"
        description: "Sum order amount for food orders."
        label: "food_order_amount"
        type: simple
        agg: sum
        expr: case when is_food_order then order_total else 0 end
      - name: "max_order_amount"
        label: "max_order_amount"
        description: "The highest order value for a given period"
        type: simple
        agg: max
        agg_time_dimension: ordered_at
        expr: order_total
      - name: "min_order_amount"
        label: "min_order_amount"
        description: "The lowest order value for a given period"
        type: simple
        agg: min
        expr: order_total
      - name: new_customer
        description: Unique count of new customers.
        label: New Customers
        type: simple
        filter: |
          {{ Dimension('customer__customer_type') }}  = 'new'
        agg: count_distinct
        expr: customer_id
      - name: "order_amount"
        label: "order_amount"
        description: The cost of fulfilling each order
        type: simple
        agg: sum
      - name: tax_paid
        description: The total tax paid on each order.
        agg: sum
        type: simple
        hidden: true
      - name: "food_order_pct"
        description: "The food cost as the % of the total order"
        label: "food_order_pct"
        type: ratio
        numerator: food_order_amount
        denominator: order_total

        type: derived
        expr: food_order_pct * 100
        input_metrics:
